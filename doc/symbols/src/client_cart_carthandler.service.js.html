 26             getItems: getItems,
 27             getItemByID: getItemByID,
 28             getLength: getLength,
 29             getArticleCount : getArticleCount,

 38         function getItems()
 39         {
 40             return items;

 42

 46         };
 47 
 48         function getLength()
 49         {
 50             return items.length;
 51         };
 52         
 53         function getArticleCount()
 54         {
 55             var count = 0;
 56             
 57             for(var Index = 0; Index < items.length; ++ Index)
 58             {
 59                 count += items[Index].quantity;
 60             }
 61             
 62             return count;
 63         };
 64 
 65         function addArticle(article) {
 66             
 67             var found = false;
 68             var indexOfItem;
 69 
 70             for (var Index = 0; Index < items.length; ++Index)
 71             {
 72                 if (items[Index].name === article.name)
 73                 {
 74                     found = true;
 75                     indexOfItem = Index;
 76                     break;
 77                 }
 78             }
 79 
 80             if (!found)
 81             {
 82                 items.push(article);
 83                 (items[items.length - 1]).quantity = 1;
 84             }
 85             else
 86             {
 87                 (items[indexOfItem]).quantity += 1;
 88             }
 89         };
 90 
 91

 94         };

101             {
102                 (items[Index].price).substr(0, (items[Index].price).length - 1);
103                 sum += (parseFloat(items[Index].price)) * items[Index].quantity;
104             }
105 
106             return sum;
107         };
