1 /**
  2  * @namespace Factories
  3  * @description Service zur Ãœbermittlung der eingegebenen Daten an die Datenbank.

  5 (function(){
  6     
  7     'use strict';
  8     
  9     angular
 10             .module('tutorialApp')
 11             .factory('adminHandler', adminHandler)
 12     
 13     adminHandler.$inject = ['$http'];
 14     
 15     function adminHandler($http){
 16         
 17         var actions = {
 18             saveImage       : saveImage,
 19             saveLocation    : saveLocation,
 20         };
 21         return actions;
 22         
 23         /////////////////////////
 24                
 25         function saveImage(fileInput, price, name){
 26        
 27             var files       = fileInput.files;
 28             var imageFile   = files[0];
 29             var reader      = new FileReader();
 30             reader.onload   = loadFunction;   
 31             reader.readAsDataURL(imageFile);
 32  
 33             /////////////////////////////////////////////////////////
 34             
 35             function successHandler(e){
 36                 console.log('POST image successful', e);
 37             };
 38             
 39             function errorHandler(e){
 40                 console.log('An error is occured! POST image failed', e);
 41             }
 42             
 43             function loadFunction(e){
 44                 
 45                 var postPizza = [];
 46                 price     = parseFloat(price);
 47                 
 48                 var pizza     = {
 49                                     name :  name,
 50                                     price:  price,
 51                                     image:  e.target.result
 52                                 };
 53                 
 54                 postPizza = JSON.stringify(pizza);
 55                     
 56                 $http.post('/saveImage', postPizza)
 57                                 .success(successHandler(e))
 58                                 .error(errorHandler(e));
 59                         
 60             }
 61             
 62         };
 63         
 64         function saveLocation(location, longitude, latitude){
 65           
 66             var coordinates = [longitude, latitude];
 67             
 68             var postLocation =  {
 69                                     location    : location,
 70                                     coordinates : coordinates
 71                                 };
 72             
 73             $http.post('/saveLocation', postLocation)
 74                     .success(successHandler)
 75                     .error(errorHandler);
 76             
 77             ////////////////////////////////////////
 78             
 79             function successHandler(){
 80                 console.log("POST saveLocation successful!");
 81             };
 82             
 83             function errorHandler(){
 84                 console.log("POST saveLocation failes!");
 85             };
 86             
 87         };
 88         
 89     }
 90     
 91 })();
